function(ADD_TOOL TOOL_NAME SRC_FILE)
    add_executable(${TOOL_NAME} ${SRC_FILE})
    target_link_libraries(${TOOL_NAME} app)
    install(TARGETS ${TOOL_NAME} DESTINATION bin)
endfunction()
